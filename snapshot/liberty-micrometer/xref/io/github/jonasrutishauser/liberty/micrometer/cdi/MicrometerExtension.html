3   import java.util.HashSet;
4   import java.util.Set;
5   import java.util.stream.Collectors;
6   
7   import io.micrometer.common.annotation.ValueExpressionResolver;
8   import io.micrometer.common.annotation.ValueResolver;
9   import io.micrometer.core.annotation.Counted;
10  import io.micrometer.core.annotation.Timed;
11  import jakarta.enterprise.event.Observes;
12  import jakarta.enterprise.inject.AmbiguousResolutionException;
13  import jakarta.enterprise.inject.spi.AfterBeanDiscovery;
14  import jakarta.enterprise.inject.spi.Bean;
15  import jakarta.enterprise.inject.spi.BeanManager;
16  import jakarta.enterprise.inject.spi.BeforeBeanDiscovery;
17  import jakarta.enterprise.inject.spi.Extension;
18  import jakarta.enterprise.inject.spi.ProcessBean;
19  import jakarta.enterprise.util.Nonbinding;
20  import jakarta.inject.Singleton;
21  
22  public class MicrometerExtension implements Extension {
23  
24      private final Set<Bean<? extends ValueResolver>> valueResolverBeans = new HashSet<>();
25      private Bean<? extends ValueExpressionResolver> valueExpressionResolverBean;
26  
27      void beforeBeanDiscovery(@Observes BeforeBeanDiscovery event) {
28          event.configureInterceptorBinding(Counted.class).methods()
29                  .forEach(member -> member.add(Nonbinding.Literal.INSTANCE));
30          event.configureInterceptorBinding(Timed.class).methods()
31                  .forEach(member -> member.add(Nonbinding.Literal.INSTANCE));
32          event.addAnnotatedType(MicrometerCountedInterceptor.class, MicrometerCountedInterceptor.class.getName())
33                  .add(new CountedLiteral());
34          event.addAnnotatedType(MicrometerTimedInterceptor.class, MicrometerTimedInterceptor.class.getName());
35      }
36  
37      <T extends ValueResolver> void discoverValueResolvers(@Observes ProcessBean<T> event) {
38          if (event.getBean().getBeanClass() != null
39                  && ValueResolver.class.isAssignableFrom(event.getBean().getBeanClass())) {
40              valueResolverBeans.add(event.getBean());
41          }
42      }
43  
44      <T extends ValueExpressionResolver> void discoverValueExpressionResolver(@Observes ProcessBean<T> event) {
45          if (event.getBean().getBeanClass() != null
46                  && ValueResolver.class.isAssignableFrom(event.getBean().getBeanClass())) {
47              if (valueExpressionResolverBean != null) {
48                  event.addDefinitionError(
49                          new AmbiguousResolutionException("Multiple ValueExpressionResolver beans found: "
50                                  + valueExpressionResolverBean + " and " + event.getBean()));
51              }
52              valueExpressionResolverBean = event.getBean();
53          }
54      }
55  
56      void registerMeterTagsSupport(@Observes AfterBeanDiscovery event, BeanManager beanManager) {
57          event.addBean() //
58                  .beanClass(MeterTagsSupport.class) //
59                  .addType(MeterTagsSupport.class) //
60                  .scope(Singleton.class) //
61                  .createWith(
62                          ctx -> new MeterTagsSupport(
63                                  valueResolverBeans.stream()
64                                          .collect(Collectors.toMap(Bean::getBeanClass,
65                                                  bean -> (ValueResolver) beanManager.getReference(bean,
66                                                          ValueResolver.class, ctx))),
67                                  valueExpressionResolverBean == null ? null
68                                          : (ValueExpressionResolver) beanManager.getReference(
69                                                  valueExpressionResolverBean, ValueExpressionResolver.class, ctx)));
70      }

72  }
