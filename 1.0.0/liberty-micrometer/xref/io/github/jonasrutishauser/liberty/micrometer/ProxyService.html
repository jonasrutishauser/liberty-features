3   import java.util.Arrays;
4   import java.util.Collection;
5   import java.util.Map;

7   import org.osgi.framework.Bundle;
8   import org.osgi.framework.BundleContext;
9   import org.osgi.framework.BundleException;
10  import org.osgi.framework.ServiceEvent;
11  import org.osgi.framework.ServiceReference;
12  import org.osgi.framework.ServiceRegistration;
13  import org.osgi.framework.hooks.service.EventListenerHook;
14  import org.osgi.framework.hooks.service.FindHook;
15  import org.osgi.framework.hooks.service.ListenerHook.ListenerInfo;
16  
17  public class ProxyService<T> implements EventListenerHook, FindHook {
18  
19      private final Bundle bundle;
20      private final ServiceReference<T> reference;
21      private final BundleContext context;
22      private ServiceRegistration<?> reg;
23  
24      ProxyService(BundleContext context, ServiceReference<T> reference, Bundle bundle) {
25          this.context = context;
26          this.bundle = bundle;
27          this.reference = reference;
28      }
29  
30      boolean open() throws BundleException {
31          boolean active = bundle.getState() == Bundle.ACTIVE;
32          if (active) {
33              bundle.stop(Bundle.STOP_TRANSIENT);
34          }
35          reg = context.registerService(new String[] {FindHook.class.getName(), EventListenerHook.class.getName()}, this,
36                  null);
37          if (active) {
38              bundle.start(Bundle.START_TRANSIENT);
39          }
40          return active;
41      }
42  
43      public void close() {
44          reg.unregister();
45      }
46  
47      @Override
48      public void find(BundleContext context, String name, String filter, boolean allServices,
49              Collection<ServiceReference<?>> references) {
50          if (context.getBundle() == bundle) {
51              references.remove(reference);
52          } else if (Arrays.asList((String[]) reference.getProperty("objectClass")).contains(name)) {
53              references.removeIf(ref -> ref.getBundle() == context.getBundle());
54          }
55      }
56  
57      @Override
58      public void event(ServiceEvent event, Map<BundleContext, Collection<ListenerInfo>> listeners) {
59          if (event.getServiceReference().equals(reference)) {
60              listeners.remove(bundle.getBundleContext());
61          }
62      }
63  
64  }
